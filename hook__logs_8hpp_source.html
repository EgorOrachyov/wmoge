56  Status on_process(CmdLine& cmd_line, class Engine& engine) override {

59  const bool no_logs = cmd_line.get_bool("disable_logs");
60  if (no_logs) {
61  return StatusCode::Ok;
62  }
63 
64  const bool log_to_out = config->get_bool(SID("engine.log_to_out"), true);
65  const bool log_to_file = config->get_bool(SID("engine.log_to_file"), true);
66  const bool log_to_console = config->get_bool(SID("engine.log_to_console"), true);
67 
68  const auto log_to_out_level = Enum::parse<LogLevel>(config->get_string(SID("engine.log_to_out_level"), "Info"));
69  const auto log_to_file_level = Enum::parse<LogLevel>(config->get_string(SID("engine.log_to_file_level"), "Info"));
70  const auto log_to_console_level = Enum::parse<LogLevel>(config->get_string(SID("engine.log_to_console_level"), "Info"));
71 
72  if (log_to_file) {
73  auto log_listener_stream = std::make_shared<LogListenerStream>("file", log_to_file_level);
74  Log::instance()->listen(log_listener_stream);
75  WG_LOG_INFO("attach file log listener");
76  }
77  if (log_to_out) {
78  auto log_listener_stdout = std::make_shared<LogListenerStdout>("out", log_to_out_level);
79  Log::instance()->listen(log_listener_stdout);
80  WG_LOG_INFO("attach stdout log listener");
81  }
82  if (log_to_console) {
83  auto log_listener_console = std::make_shared<LogListenerConsole>(engine.console(), log_to_console_level);
84  Log::instance()->listen(log_listener_console);
85  WG_LOG_INFO("attach console log listener");
86  }
87 
88  return StatusCode::Ok;
89  }
