86  Status tree_read(IoContext& context, IoTree& tree, AssetRef<T>& ref) {
87  Ref<Asset> asset_ref;
88  WG_CHECKED(AssetRefParser::parse_from_tree(context, tree, asset_ref));
89  if (asset_ref) {
