117  template<typename E, typename C>
118  EventListenerHnd EventManager::subscribe(C&& callable) {
119  EventCallback callback = [c = std::forward<C>(callable)](const Ref<Event>& event) {
120  const E* casted_event = dynamic_cast<const E*>(event.get());
121  if (!casted_event) {
122  WG_LOG_ERROR("failed to cast event to expected type");
123  return false;
124  }
125  return c(*casted_event);
126  };
127  return subscribe(E::type_static(), std::move(callback));
128  }
