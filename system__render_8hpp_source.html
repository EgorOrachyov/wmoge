53  void process(EcsWorld& world, const EcsEntity& enity, EcsComponentLocalToWorld& l2w, EcsComponentMeshStatic& mesh_static) {
54  if (mesh_static.dirty) {
55  const int id = mesh_static.mesh->get_primitive_id();
56  const Mat4x4f mat = l2w.matrix.transpose();
57  const Aabbf aabb = mesh_static.mesh->get_aabb().transform(l2w.matrix);
58 
59  vis_system->update_item_bbox(mesh_static.vis_item, aabb);
60 
61  GPURenderObjectData& gpu_data = render_scene->get_objects_gpu_data()[id];
62  gpu_data.LocalToWorld = mat;
63  gpu_data.LocalToWorldPrev = mat;
64  gpu_data.NormalMatrix = mat;
65  gpu_data.AabbPos = Vec4f(aabb.center(), 0.0f);
66  gpu_data.AabbSizeHalf = Vec4f(aabb.extent(), 0.0f);
67 
68  mesh_static.dirty = false;
69  }
70  }
