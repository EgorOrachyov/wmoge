390  static TVecN cross(const TVecN& a, const TVecN& b) {
391  if constexpr (N == 3) {
392  TVecN<T, 3> result;
393 
394  // a x b = det | i j k |
395  // | x1 y1 z1 |
396  // | x2 y2 z2 |
397 
398  result[0] = a.y() * b.z() - a.z() * b.y();
399  result[1] = a.z() * b.x() - a.x() * b.z();
400  result[2] = a.x() * b.y() - a.y() * b.x();
401 
402  return result;
403  } else {
404  return TVecN();
405  }
406  }

434  static TVecN slerp(T t, T ang, const TVecN& a, const TVecN& b) {
435  if (ang <= Math::THRESH_FLOAT32) {
436  return lerp(t, a, b);
437  }
438 
439  TVecN<T, N> r;
440  const T angle_sin = Math::sin(ang);
441  const T angle1 = Math::sin(ang * (1 - t)) / angle_sin;
442  const T angle2 = Math::sin(ang * t) / angle_sin;
443 
444  for (int i = 0; i < N; i++) {
445  r.values[i] = a.values[i] * angle1 + b.values[i] * angle2;
446  }
447 
448  return r;
