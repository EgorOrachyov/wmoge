69  std::cerr << "failed to load engine engine.cfg file, check your configuration file or path";
70  }
71  if (!config->load(config_path + "/game.cfg", ConfigStackMode::Keep)) {
72  std::cerr << "failed to load engine game.cfg file, check your configure file of path";
73  }
