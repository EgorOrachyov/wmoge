56  Status on_process(CmdLine& cmd_line) override {

60 
61  enable_profiler = enable_profiler || cmd_line.get_bool("profiler");
62  enable_profiler = enable_profiler || config->get_bool(SID("debug.profiler"), false);
63 
64  profiler->set_enabled(enable_profiler);
65 
66  if (enable_profiler) {
67  WG_LOG_INFO("attach cpu performance profiler");

73  WG_PROFILE_CAPTURE_END(m_capture_startup);
