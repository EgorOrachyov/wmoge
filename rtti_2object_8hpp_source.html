66  template<typename T, typename std::enable_if<std::is_convertible_v<T*, RttiObject*>>::type>
67  Status copy_objects(const std::vector<Ref<T>>& objects, std::vector<Ref<T>>& copied) {
68  static_assert(std::is_base_of_v<RttiObject, T>, "T must be sub-class of object");
69 
70  copied.reserve(objects.size());
71 
72  for (auto& object : objects) {
73  Ref<RttiObject> copy;
74 
75  if (!object->clone(copy)) {
76  WG_LOG_ERROR("failed to clone object " << object->class_name());
77  return StatusCode::Error;
78  }
79 
80  auto as_t = copy.cast<T>();
81  assert(as_t);
82 
83  copied.push_back(as_t);
84  }
85 
86  return StatusCode::Ok;
87  }
