85  virtual void add_on_completion(AsyncCallback<T> callback) {
86  if (!callback) return;
87  std::lock_guard lock(m_mutex);
88  if (is_completed()) {
89  callback(m_status.load(), m_result);
90  return;
91  }
92  m_callbacks.push_back(std::move(callback));
93  }
94  virtual void set_failed() {
95  std::lock_guard lock(m_mutex);
96  m_status.store(AsyncStatus::Failed);
97 
98  buffered_vector<AsyncCallback<T>> callbacks = std::move(m_callbacks);
99  for (auto& c : callbacks) c(AsyncStatus::Failed, m_result);
100  buffered_vector<Ref<AsyncStateBase>> deps = std::move(m_children);
101  for (auto& d : deps) d->notify(AsyncStatus::Failed, this);
102  }
103  virtual void set_result(T&& result) {
104  std::lock_guard lock(m_mutex);
105  m_result = std::make_optional(std::forward<T>(result));
106  m_status.store(AsyncStatus::Ok);
107 
108  buffered_vector<AsyncCallback<T>> callbacks = std::move(m_callbacks);
109  for (auto& c : callbacks) c(AsyncStatus::Ok, m_result);
110  buffered_vector<Ref<AsyncStateBase>> deps = std::move(m_children);
111  for (auto& d : deps) d->notify(AsyncStatus::Ok, this);
112  }
113 
114  void add_dependency(const Ref<AsyncStateBase>& dependency) override {
115  assert(dependency);
116  std::lock_guard lock(m_mutex);
117  if (is_completed()) {
118  dependency->notify(status(), this);
119  return;
120  }
121  m_children.push_back(dependency);
122  }

124  void wait_completed() override {
