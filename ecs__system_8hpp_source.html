131  WG_AUTO_PROFILE_ECS_DECS("EcsSystemBindHelper::process_batch", m_system_ptr->get_name().str());
132 
133  static const auto num_args = sizeof...(TArgs);
134 
135  for (int i = 0; i < count; i++) {
136  const int entity_idx = start_entity + i;
137  const EcsEntity entity = storage.get_entity(entity_idx);
138 
139  if (entity.is_invalid()) {
140  continue;
141  }

143 #define WG_ECS_ACCESS_COMPONENT(at) \
144  using TypeComponent##at = typename std::remove_reference<typename std::tuple_element<at, std::tuple<TArgs...>>::type>::type; \

146 
147  if constexpr (num_args == 0) {
