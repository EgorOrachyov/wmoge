135  void EcsSystemBindHelper<T, TArgs...>::process_batch(class EcsWorld& world, EcsArchStorage& storage, int start_entity, int count) {
136  WG_AUTO_PROFILE_ECS_DECS("EcsSystemBindHelper::process_batch", m_system_ptr->get_name().str());
137 
138  static const auto num_args = sizeof...(TArgs);
139 
140  for (int i = 0; i < count; i++) {
141  const int entity_idx = start_entity + i;
142  const EcsEntity entity = storage.get_entity(entity_idx);
143 
144  assert(entity.is_valid());
145 
146 #define WG_ECS_ACCESS_COMPONENT(at) \
147  using TypeComponent##at = typename std::remove_reference<typename std::tuple_element<at, std::tuple<TArgs...>>::type>::type; \
148  TypeComponent##at* ptr##at = storage.template get_component<TypeComponent##at>(entity_idx);
149 
150  if constexpr (num_args == 0) {
151  (*m_system_ptr.*m_function_ptr)(world, entity);
152  } else if constexpr (num_args == 1) {
153  WG_ECS_ACCESS_COMPONENT(0);
154  (*m_system_ptr.*m_function_ptr)(world, entity, *ptr0);
155  } else if constexpr (num_args == 2) {
156  WG_ECS_ACCESS_COMPONENT(0);
157  WG_ECS_ACCESS_COMPONENT(1);
158  (*m_system_ptr.*m_function_ptr)(world, entity, *ptr0, *ptr1);
159  } else if constexpr (num_args == 3) {
160  WG_ECS_ACCESS_COMPONENT(0);
161  WG_ECS_ACCESS_COMPONENT(1);
162  WG_ECS_ACCESS_COMPONENT(2);
163  (*m_system_ptr.*m_function_ptr)(world, entity, *ptr0, *ptr1, *ptr2);
164  } else if constexpr (num_args == 4) {
165  WG_ECS_ACCESS_COMPONENT(0);
166  WG_ECS_ACCESS_COMPONENT(1);
167  WG_ECS_ACCESS_COMPONENT(2);

220  EcsQuery query;

222  std::tuple<std::remove_reference_t<TArgs>...> pack;
223  constexpr std::size_t N = std::tuple_size_v<decltype(pack)>;
224 
225 #define WG_ECS_PROCESS_ARG(id) \
226  if constexpr (id < N) { \
227  using Arg = std::tuple_element_t<0, decltype(pack)>; \
228  using NoConst = std::remove_const_t<Arg>; \
229  if constexpr (std::is_same_v<Arg, NoConst>) { \
230  query.set_read<NoConst>(); \
231  } else { \
232  query.set_write<NoConst>(); \
233  } \
234  }
