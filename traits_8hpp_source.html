71  template<typename T>
72  inline RttiType* rtti_type() {
73  const Strid rtti_name = SID(RttiTypeOf<T>::name());
74  static RttiType* g_type = RttiTypeStorage::instance()->find_type(rtti_name).value_or(nullptr);
75 
76  if (!g_type) {
77  g_type = RttiTypeStorage::instance()->find_type(rtti_name).value_or(nullptr);
78 
79  if (!g_type) {
80  Ref<RttiType> type = RttiTypeOf<T>::make();
81  RttiTypeStorage::instance()->add(type);
82  g_type = type.get();
83  }
84  }
85 
86  return g_type;
87  }

597  Status call(RttiFrame& frame, class RttiObject* p_object, void* p_args_ret) override {
598  using TArgsTuple = std::tuple<TArgs...>;
599 
600  constexpr bool has_return = !std::is_void<TRet>::value;
601  constexpr auto n_args = std::tuple_size_v<TArgsTuple>;
602 
603  TClass* target = (TClass*) p_object;
604 
605 #define OFFSET_OF(info) \
606  ((std::uint8_t*) p_args_ret + info.stack_offset)
607 
608 #define GET_ARG(index) *([&]() { \
609  using tinfo = RttiArgInfo<std::tuple_element_t<index, TArgsTuple>>; \
610  if constexpr (tinfo::is_ref) { \
611  return *((typename tinfo::BaseType**) OFFSET_OF(get_args()[index])); \
612  } else if constexpr (tinfo::is_ptr) { \
613  return ((typename tinfo::BaseType**) OFFSET_OF(get_args()[index])); \
614  } else { \
615  return ((typename tinfo::BaseType*) OFFSET_OF(get_args()[index])); \
616  } \
617 }())
618 
619 #define SET_RET(r) \
620  using tinfo = RttiArgInfo<TRet>; \
621  if constexpr (tinfo::is_ref) { \
622  *((typename tinfo::BaseType**) OFFSET_OF(get_ret())) = &r; \
623  } else if constexpr (tinfo::is_ptr) { \
624  *((typename tinfo::BaseType**) OFFSET_OF(get_ret())) = r; \
625  } else { \
626  *((typename tinfo::BaseType*) OFFSET_OF(get_ret())) = std::move(r); \
627  }
628 
629 #define DO_CALL(...) \
630  if constexpr (has_return) { \
631  TRet r = (*target.*m_p_method)(__VA_ARGS__); \
632  SET_RET(r); \
633  } else { \
634  (*target.*m_p_method)(__VA_ARGS__); \
635  }
636 
637  if constexpr (n_args == 0) {
638  DO_CALL();
639  } else if constexpr (n_args == 1) {
640  DO_CALL(GET_ARG(0));
641  } else if constexpr (n_args == 2) {
642  DO_CALL(GET_ARG(0), GET_ARG(1));
643  } else if constexpr (n_args == 3) {
644  DO_CALL(GET_ARG(0), GET_ARG(1), GET_ARG(2));
645  } else if constexpr (n_args == 4) {
646  DO_CALL(GET_ARG(0), GET_ARG(1), GET_ARG(2), GET_ARG(3));
647  } else if constexpr (n_args == 5) {
648  DO_CALL(GET_ARG(0), GET_ARG(1), GET_ARG(2), GET_ARG(3), GET_ARG(4));
649  } else {
650  WG_LOG_ERROR("too much arguments count");
651  return StatusCode::NotImplemented;
652  }
653 
654  return WG_OK;
655 
656 #undef OFFSET_OF
657 #undef GET_ARG
658 #undef SET_RET
659 #undef DO_CALL
660  }

663  TSignature m_p_method;

687  template<typename TSignature, typename TRet, typename... TArgs>
688  void add_method_t(Strid name, TSignature p_method, const buffered_vector<const char*>& names, RttiMetaData meta_data) {
689  using TArgsTuple = std::tuple<TArgs...>;
690 
691  constexpr bool has_return = !std::is_void<TRet>::value;
692  constexpr auto n_args = std::tuple_size_v<TArgsTuple>;
693  constexpr auto n_names = has_return ? 1 + n_args : n_args;
694 
695  static_assert(n_args <= MAX_FUCNTION_ARGS, "too much arguments");
696 
697  if (n_names != n_args) {
698  WG_LOG_ERROR("mismatched args names size for" << get_name());
699  return;
700  }
701 
702 #define PARAM_INFO(param, str_name) \
703  param.type = rtti_type<typename tinfo::BaseType>(); \
704  param.name = SID(str_name); \
705  param.is_const = tinfo::is_const; \
706  param.is_ref = tinfo::is_ref; \
707  param.is_ptr = tinfo::is_ptr; \
708  param.stack_offset = stack_size; \
709  param.stack_size = param.is_ptr || param.is_ref ? sizeof(void*) : param.type->get_byte_size(); \
710  stack_size += param.stack_size;
711 
712 #define ARG_INFO(index) \
713  if constexpr (index < n_args) { \
714  using tinfo = RttiArgInfo<std::tuple_element_t<index, TArgsTuple>>; \
715  RttiParamInfo& param = args.emplace_back(); \
716  PARAM_INFO(param, names[index]) \
717  }
718 
719  RttiParamInfo ret;
720  std::vector<RttiParamInfo> args;
721  std::size_t stack_size = 0;
722 
723  ARG_INFO(0);
724  ARG_INFO(1);
725  ARG_INFO(2);
726  ARG_INFO(3);
727  ARG_INFO(4);
728 
729  if constexpr (has_return) {
730  using tinfo = RttiArgInfo<TRet>;
731  PARAM_INFO(ret, names.back())
732  }
733 
734  std::stringstream method_name;
735 
736  auto print_param = [&](const RttiParamInfo& p) {
737  if (!p.type) {
738  method_name << "void";
739  return;
740  }
741 
742  if (p.is_const) {
743  method_name << "const ";
744  }
745 
746  method_name << p.type->get_str();
747 
748  if (p.is_ptr) {
749  method_name << "*";
750  }
751  if (p.is_ref) {
752  method_name << "&";
753  }
754 
755  method_name << " ";
756  method_name << p.name.str();
757  };
758 
759  print_param(ret);
760  method_name << " " << get_name().str() << "::";
761  method_name << name.str();
762  method_name << "(";
763  for (const auto& arg : args) {
764  print_param(arg);
765  method_name << ",";
766  }
767  method_name << ")";
768 
769  RttiMethod method(name, make_ref<RttiFunctionT<ClassT, TSignature, TRet, TArgs...>>(Strid(method_name.str()), p_method, stack_size, std::move(args), std::move(ret)));
770  method.set_metadata(std::move(meta_data));
771  RttiClass::add_method(std::move(method));
772 
773 #undef PARAM_INFO
774 #undef ARG_INFO
775  }
776  template<typename TRet, typename... TArgs>
777  void add_method(Strid name, TRet (ClassT::*p_method)(TArgs...), const buffered_vector<const char*>& names, RttiMetaData meta_data) {
778  add_method_t<decltype(p_method), TRet, TArgs...>(name, p_method, names, std::move(meta_data));
779  }

782  add_method_t<decltype(p_method), TRet, TArgs...>(name, p_method, names, std::move(meta_data));
