369  template<typename Performer>
370  Status perform_call(Performer& performer, RttiFrame& frame, void* target, array_view<std::uint8_t> args_ret) const {
371  using TArgsTuple = std::tuple<TArgs...>;
372 
373  constexpr bool has_return = !std::is_void<TRet>::value;
374  constexpr auto n_args = std::tuple_size_v<TArgsTuple>;
375 
376  void* p_args_ret = args_ret.data();
377 
378  if (!target) {
379  WG_LOG_ERROR("null target passed to call " << RttiTypeFunction::get_name());
380  return StatusCode::InvalidParameter;
381  }
382  if (args_ret.size() != m_stack_size) {
383  WG_LOG_ERROR("invalid args_ret passed to call " << RttiTypeFunction::get_name());
384  return StatusCode::InvalidParameter;
385  }
386 
387 #define OFFSET_OF(info) \
388  ((std::uint8_t*) p_args_ret + info.stack_offset)
389 
390 #define GET_ARG(index) *([&]() { \
391  using tinfo = RttiArgInfo<std::tuple_element_t<index, TArgsTuple>>; \
392  if constexpr (tinfo::is_ref) { \
393  return *((typename tinfo::BaseType**) OFFSET_OF(get_args()[index])); \
394  } else if constexpr (tinfo::is_ptr) { \
395  return ((typename tinfo::BaseType**) OFFSET_OF(get_args()[index])); \
396  } else { \
397  return ((typename tinfo::BaseType*) OFFSET_OF(get_args()[index])); \
398  } \
399 }())
400 
401 #define SET_RET(r) \
402  using tinfo = RttiArgInfo<TRet>; \
403  if constexpr (tinfo::is_ref) { \
404  *((typename tinfo::BaseType**) OFFSET_OF(get_ret())) = &r; \
405  } else if constexpr (tinfo::is_ptr) { \
406  *((typename tinfo::BaseType**) OFFSET_OF(get_ret())) = r; \
407  } else { \
408  *((typename tinfo::BaseType*) OFFSET_OF(get_ret())) = std::move(r); \
409  }
410 
411 #define DO_CALL(...) \
412  do { \
413  if constexpr (has_return) { \
414  TRet r = performer(__VA_ARGS__); \
415  SET_RET(r); \
416  } else { \
417  performer(__VA_ARGS__); \
418  } \
419  } while (0)
420 
421  if constexpr (n_args == 0) {
422  DO_CALL(target);
423  } else if constexpr (n_args == 1) {
424  DO_CALL(target, GET_ARG(0));
425  } else if constexpr (n_args == 2) {
426  DO_CALL(target, GET_ARG(0), GET_ARG(1));
427  } else if constexpr (n_args == 3) {
428  DO_CALL(target, GET_ARG(0), GET_ARG(1), GET_ARG(2));
429  } else if constexpr (n_args == 4) {
430  DO_CALL(target, GET_ARG(0), GET_ARG(1), GET_ARG(2), GET_ARG(3));
431  } else if constexpr (n_args == 5) {
432  DO_CALL(target, GET_ARG(0), GET_ARG(1), GET_ARG(2), GET_ARG(3), GET_ARG(4));
433  } else {
434  WG_LOG_ERROR("too much arguments count");
435  return StatusCode::NotImplemented;
436  }
437 
438  return WG_OK;
439 
440 #undef OFFSET_OF
441 #undef GET_ARG
442 #undef SET_RET
443 #undef DO_CALL
444  }

446  static void fill_param_info(std::vector<RttiParamInfo>& args, RttiParamInfo& ret, std::size_t& stack_size) {
447  using TArgsTuple = std::tuple<TArgs...>;
448 
449  constexpr bool has_return = !std::is_void<TRet>::value;
450  constexpr auto n_args = std::tuple_size_v<TArgsTuple>;
451 
452  static_assert(n_args <= MAX_ARGS_COUNT, "too much arguments");
453 
454 #define PARAM_INFO(param) \
455  param.type = rtti_type<typename tinfo::BaseType>(); \
456  param.is_const = tinfo::is_const; \
457  param.is_ref = tinfo::is_ref; \
458  param.is_ptr = tinfo::is_ptr; \
459  param.stack_offset = stack_size; \
460  param.stack_size = param.is_ptr || param.is_ref ? sizeof(void*) : param.type->get_byte_size(); \
461  stack_size += param.stack_size;
462 
463 #define ARG_INFO(index) \
464  if constexpr (index < n_args) { \
465  using tinfo = RttiArgInfo<std::tuple_element_t<index, TArgsTuple>>; \
466  RttiParamInfo& param = args.emplace_back(); \
467  PARAM_INFO(param) \
468  }
469 
470  ARG_INFO(0);
471  ARG_INFO(1);
472  ARG_INFO(2);
473  ARG_INFO(3);
474  ARG_INFO(4);
475 
476  if constexpr (has_return) {
477  using tinfo = RttiArgInfo<TRet>;
478  PARAM_INFO(ret)
479  }
480 
481 #undef PARAM_INFO
482 #undef ARG_INFO
483  }
