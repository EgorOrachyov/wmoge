58  template<typename T>
59  inline RttiType* rtti_type() {
60  static RttiType* g_type = RttiTypeStorage::instance()->find_type(SID(RttiTypeOf<T>::name())).value_or(nullptr);
61 
62  if (!g_type) {
63  Ref<RttiType> type = RttiTypeOf<T>::make();
64  RttiTypeStorage::instance()->add(type);
65  g_type = type.get();
66  }
67 
68  return g_type;
69  }
