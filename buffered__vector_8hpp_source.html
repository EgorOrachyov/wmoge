80  Status yaml_read(IoContext& context, YamlConstNodeRef node, buffered_vector<T, MinCapacity>& vector) {
81  assert(vector.empty());
82  vector.resize(node.num_children());
83  std::size_t element_id = 0;
84  for (auto child = node.first_child(); child.valid(); child = child.next_sibling()) {
85  WG_YAML_READ(context, child, vector[element_id]);
86  element_id += 1;
87  }
88  return StatusCode::Ok;
