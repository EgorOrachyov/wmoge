163  template<typename T, typename... Args>
164  Class* Class::add_method(wmoge::Method method, void (T::*p_method)(Args...), const std::vector<Var>& defaults) {
165  std::vector<Var> args_values(method.args_count());
166  std::copy(defaults.begin(), defaults.end(), args_values.begin() + (method.args_count() - defaults.size()));
167  method.m_args_values = std::move(args_values);
168  method.m_callable = [=](const Method& m, Object* obj, int argc, const Var* argv, Var&) -> int {
169  assert(obj);
170  static_assert(sizeof...(Args) <= 4, "methods args count support is limited");
171 
172  const std::size_t to_fill = sizeof...(Args);
173  const Var* argvp[to_fill + 1];
174  const std::vector<Var>& defaults = method.args_values();
175  std::size_t it = 0;
176 
177  while (it < to_fill && argc > 0) {
178  argvp[it] = &argv[it];
179  it += 1;
180  argc -= 1;
181  }
182  while (it < to_fill) {
183  argvp[it] = &defaults[it];
184  it += 1;
185  }
186 
187  T* target = (T*) obj;
188 
189  if constexpr (sizeof...(Args) == 0)
190  (*target.*p_method)();
191  else if constexpr (sizeof...(Args) == 1)
192  (*target.*p_method)(*argvp[0]);
193  else if constexpr (sizeof...(Args) == 2)
194  (*target.*p_method)(*argvp[0], *argvp[1]);
195  else if constexpr (sizeof...(Args) == 3)
196  (*target.*p_method)(*argvp[0], *argvp[1], *argvp[2]);
197  else if constexpr (sizeof...(Args) == 4)
198  (*target.*p_method)(*argvp[0], *argvp[1], *argvp[2], *argvp[3]);
199 
200  return 0;
201  };
202  m_methods.emplace(method.name(), std::move(method));
203  return this;
204  }
