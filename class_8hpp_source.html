217  template<typename T, typename... Args>
218  Class* Class::add_method(wmoge::ClassMethod method, void (T::*p_method)(Args...), const std::vector<Var>& defaults) {
219  std::vector<Var> args_values(method.args_count());
220  std::copy(defaults.begin(), defaults.end(), args_values.begin() + int(method.args_count() - defaults.size()));
221 
222  method.m_args_values = std::move(args_values);
223  method.m_callable = [=](const ClassMethod& m, Object* obj, int argc, const Var* argv, Var&) -> int {
224  assert(obj);
225  static_assert(sizeof...(Args) <= 4, "methods args count support is limited");
226 
227  const std::size_t to_fill = sizeof...(Args);
228  const Var* argvp[to_fill + 1];
229  const std::vector<Var>& defaults = method.args_values();
230  std::size_t it = 0;
231 
232  while (it < to_fill && argc > 0) {
233  argvp[it] = &argv[it];
234  it += 1;
235  argc -= 1;
236  }
237  while (it < to_fill) {
238  argvp[it] = &defaults[it];
239  it += 1;
240  }
241 
242  T* target = (T*) obj;
243 
244  if constexpr (sizeof...(Args) == 0)
245  (*target.*p_method)();
246  else if constexpr (sizeof...(Args) == 1)
247  (*target.*p_method)(*argvp[0]);
248  else if constexpr (sizeof...(Args) == 2)
249  (*target.*p_method)(*argvp[0], *argvp[1]);
250  else if constexpr (sizeof...(Args) == 3)
251  (*target.*p_method)(*argvp[0], *argvp[1], *argvp[2]);
252  else if constexpr (sizeof...(Args) == 4)
253  (*target.*p_method)(*argvp[0], *argvp[1], *argvp[2], *argvp[3]);
254 
255  return 0;
256  };
257 
258  auto ptr_method = std::make_shared<ClassMethod>(std::move(method));
259  m_members.push_back(ptr_method);
260  m_methods.emplace(ptr_method->name(), ptr_method.get());
261 
262  return this;
263  }
