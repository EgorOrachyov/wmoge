76  TFrustum(Vec pos, Vec dir, Vec up, T fov, T aspect, T near, T far) {
77  const float tan_fh = Math::tan(fov * 0.5f);
78  const float hnear_h = tan_fh * near;
79  const float hnear_w = hnear_h * aspect;
80  const float hfar_h = tan_fh * far;
81  const float hfar_w = hfar_h * aspect;
82 
83  up = up.normalized();
84  dir = dir.normalized();
85  auto right = Vec3f::cross(dir, up).normalized();
86  up = Vec3f::cross(right, dir).normalized();
87 
88  const Vec3f p_near_left_up = pos + dir * near + right * (-hnear_w) + up * hnear_h;
89  const Vec3f p_near_left_down = pos + dir * near + right * (-hnear_w) + up * (-hnear_h);
90  const Vec3f p_near_right_up = pos + dir * near + right * hnear_w + up * hnear_h;
91  const Vec3f p_near_right_down = pos + dir * near + right * hnear_w + up * (-hnear_h);
92  const Vec3f p_far_left_up = pos + dir * far + right * (-hfar_w) + up * hfar_h;
93  const Vec3f p_far_left_down = pos + dir * far + right * (-hfar_w) + up * (-hfar_h);
94  const Vec3f p_far_right_up = pos + dir * far + right * hfar_w + up * hfar_h;
95  const Vec3f p_far_right_down = pos + dir * far + right * hfar_w + up * (-hfar_h);
96 
97  planes[0] = Planef(p_near_left_up, p_near_left_down, p_far_left_down); // Left
98  planes[1] = Planef(p_near_right_up, p_far_right_down, p_near_right_down); // Right
99  planes[2] = Planef(p_near_left_down, p_near_right_down, p_far_right_down);// Bottom
100  planes[3] = Planef(p_near_right_up, p_near_left_up, p_far_right_up); // Top
101  planes[4] = Planef(p_near_right_up, p_near_right_down, p_near_left_down); // Near
102  planes[5] = Planef(p_far_left_up, p_far_left_down, p_far_right_down); // Far
103 
104  points[0] = p_near_left_up;
105  points[1] = p_near_left_down;
106  points[2] = p_near_right_up;
107  points[3] = p_near_right_down;
108  points[4] = p_far_left_up;
109  points[5] = p_far_left_down;
110  points[6] = p_far_right_up;
111  points[7] = p_far_right_down;
112  }

114  [[nodiscard]] bool is_inside_or_intersects(const Aabbf& box) const {
115  const auto c = box.center();
116  const auto e = box.extent();
117 
118  for (const auto& p : planes) {
119  const auto r = Vec3f::dot(e, p.norm.abs());
120  const auto s = p.dot(c);
121 
122  if (s < -r) return false;
123  }
124 
125  return true;
126  }
