92  Status copy_objects(const std::vector<Ref<T>>& objects, std::vector<Ref<T>>& copied) {
93  static_assert(std::is_base_of_v<Object, T>, "T must be sub-class of object");
94 
95  copied.reserve(objects.size());
96 
97  for (auto& object : objects) {
98  Ref<Object> copy;
99 
100  if (!object->clone(copy)) {
101  WG_LOG_ERROR("failed to clone object " << object->class_name());
102  return StatusCode::Error;
103  }
104 
105  auto as_t = copy.cast<T>();
106  assert(as_t);
107 
108  copied.push_back(as_t);
109  }
110 
111  return StatusCode::Ok;
112  }
