46  static InputModifiers mods_mask(int mods) {

48 
49  if (mods & GLFW_MOD_ALT)
50  mask.set(InputModifier::Alt);
51  if (mods & GLFW_MOD_CAPS_LOCK)
52  mask.set(InputModifier::CapsLock);

54  mask.set(InputModifier::Control);
55  if (mods & GLFW_MOD_NUM_LOCK)
56  mask.set(InputModifier::NumLock);
57  if (mods & GLFW_MOD_SHIFT)
58  mask.set(InputModifier::Shift);

60  return mask;
