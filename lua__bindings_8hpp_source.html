52  template<typename Enum>
53  static void bind_enum(luabridge::Namespace& ns, const char* name) {
54  constexpr auto entries = magic_enum::enum_entries<Enum>();
55  ns = ns.beginNamespace(name);
56 
57  for (const auto& entry : entries) {
58  auto entry_name = std::string(entry.second);
59  ns = ns.addConstant(entry_name.c_str(), static_cast<int>(entry.first));
60  }
61 
62  ns = ns.endNamespace();
63  }
64  };
