133  template<typename S>
134  inline std::optional<S*> IocContainer::resolve() {
135  auto opt_entry = get(typeid(S));
136 
137  if (!opt_entry) {
138  WG_LOG_ERROR("no such entry to resolve with type " << std::type_index(typeid(S)).name());
139  return std::nullopt;
140  }
141 
142  IocEntry& entry = *(opt_entry.value());
143 
144  if (entry.type != IocEntryType::Singleton) {
145  WG_LOG_ERROR("non-singleton entry not supported for type " << std::type_index(typeid(S)).name());
146  return std::nullopt;
147  }
148 
149  if (!entry.instance.has_value()) {
150  auto status = entry.factory(entry.instance);
151  if (!status) {
152  WG_LOG_ERROR("failed to instantiate object of type " << entry.provided_type.value().name());
153  return std::nullopt;
154  }
155  }
156 
157  S* ptr = nullptr;
158 
159  try {
160  ptr = std::any_cast<std::shared_ptr<S>>(entry.instance).get();
161  } catch (const std::bad_any_cast&) {
162  WG_LOG_ERROR("failed to cast entry to type " << std::type_index(typeid(S)).name()
163  << " from type " << entry.provided_type.value().name());
164  assert(false);
165  }
166 
167  if (!ptr) {
168  WG_LOG_ERROR("instance is null " << std::type_index(typeid(S)).name());
169  return std::nullopt;
170  }
171 
172  return ptr;
173  }
