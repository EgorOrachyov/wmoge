116

121 
122  ShaderParamInfo* param = p_info.value();
123  Ref<Data>* buffer = nullptr;
124 
125  if (param->buffer != -1) {
126  buffer = provider.get_buffer(param->space, param->buffer);
127  if (!buffer) {
128  WG_LOG_ERROR("no such buffer");

131

133  const std::int16_t size = param->elem_count * param->type->byte_size;
134  std::uint8_t* ptr = (*buffer)->buffer();
