95  Status yaml_read(YamlConstNodeRef node, TypedArray<T>& array) {
96  array.resize(node.num_children());
97  std::size_t element_id = 0;
98  for (auto child = node.first_child(); child.valid(); child = child.next_sibling()) {
99  WG_YAML_READ(child, array[element_id]);
100  element_id += 1;
101  }
102 
103  return StatusCode::Ok;
104  }
