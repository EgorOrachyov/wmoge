147  template<typename T, typename Storage>
148  void GfxVector<T, Storage>::buffer_reserve(std::size_t elements) {
149  const std::size_t capacity = buffer_capacity();
150 
151  if (capacity < elements) {
152  const std::size_t new_capacity = Math::ge_pow2_val(elements);
153  const int new_size = int(sizeof(T) * new_capacity);
154 
155  const Strid buffer_name = SID(m_name.str() + " cap=" + StringUtils::from_int(int(new_capacity)));
156 
157  if constexpr (std::is_same_v<Storage, GfxVertBuffer>) {
158  m_buffer = Engine::instance()->gfx_driver()->make_vert_buffer(new_size, GfxMemUsage::GpuLocal, buffer_name);
159  }
160  if constexpr (std::is_same_v<Storage, GfxIndexBuffer>) {
161  m_buffer = Engine::instance()->gfx_driver()->make_index_buffer(new_size, GfxMemUsage::GpuLocal, buffer_name);
162  }
163  if constexpr (std::is_same_v<Storage, GfxStorageBuffer>) {
164  m_buffer = Engine::instance()->gfx_driver()->make_storage_buffer(new_size, GfxMemUsage::GpuLocal, buffer_name);
165  }
166 
167  assert(m_buffer);
168  }
169  }
