96  Status copy_objects(const std::vector<Ref<T>>& objects, std::vector<Ref<T>>& copied) {
97  static_assert(std::is_base_of_v<Object, T>, "T must be sub-class of object");
98 
99  copied.reserve(objects.size());
100 
101  for (auto& object : objects) {
102  Ref<Object> copy;
103 
104  if (!object->clone(copy)) {
105  WG_LOG_ERROR("failed to clone object " << object->class_name());
106  return StatusCode::Error;
107  }
108 
109  auto as_t = copy.cast<T>();
110  assert(as_t);
111 
112  copied.push_back(as_t);
113  }
114 
115  return WG_OK;
116  }
