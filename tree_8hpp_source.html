247  std::size_t element_id = 0;
248  assert(tree.node_num_children() <= S);
249  tree.node_find_first_child();
250  for (; tree.node_is_valid() && element_id < S; tree.node_next_sibling()) {
251  WG_TREE_READ(context, tree, array[element_id]);

253  }
254  return WG_OK;
