188  template<typename T, int NodeCapacity>
189  inline void UnrolledList<T, NodeCapacity>::reserve(std::size_t size) {
190  if (m_size >= size) {
191  return;
192  }
193 
194  const std::size_t required_nodes_count = (size + NODE_CAPACITY - 1) / NODE_CAPACITY;
195  m_nodes.reserve(required_nodes_count);
196  m_nodes_size.reserve(required_nodes_count);
197 
198  for (std::size_t node_idx = m_nodes.size(); node_idx < required_nodes_count; node_idx++) {
199  m_nodes.push_back(static_cast<T*>(allocate_pool()));
200  m_nodes_size.push_back(0);
201  }
202  }

204  template<typename T, int NodeCapacity>
205  inline void* UnrolledList<T, NodeCapacity>::allocate_back() {
206  if (m_nodes.empty() || m_nodes_size.back() == NODE_CAPACITY) {
207  m_nodes.push_back(static_cast<T*>(allocate_pool()));
208  m_nodes_size.push_back(0);
209  }
210 
211  m_size += 1;
212  const int offset = m_nodes_size.back()++;
213  return m_nodes.back() + offset;
214  }

216  template<typename T, int NodeCapacity>
217  inline void* UnrolledList<T, NodeCapacity>::allocate_pool() {
218  if (m_pool.empty()) {
