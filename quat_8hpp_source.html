67  explicit TQuat(const Mat& mat) {
68  T q[4];// notation: x[0] y[1] z[2] w[3]
69  auto& v = mat.values;
70 
71  T trace = v[0] + v[5] + v[10];
72 
73  // Matrix 4x4 indexes
74  // 0 1 2 3
75  // 4 5 6 7
76  // 8 9 10 11
77  // 12 13 14 15
78 
79  // Check the diagonal
80  if (trace > 0.0) {
81  // positive diagonal
82 
83  T s = Math::sqrt(trace + 1.0f);
84  q[3] = static_cast<T>(s * 0.5);
85 
86  T t = 0.5f / s;
87  q[0] = (v[9] - v[6]) * t;
88  q[1] = (v[2] - v[8]) * t;
89  q[2] = (v[4] - v[1]) * t;
90  } else {
91  // negative diagonal
92 
93  int i = 0;
94  if (v[5] > v[0]) i = 1;
95  if (v[10] > v[4 * i + i]) i = 2;
96 
97  static const int NEXT[3] = {1, 2, 0};
98  int j = NEXT[i];
99  int k = NEXT[j];
100 
101  auto s = (T) Math::sqrt(v[i * 4 + i] - (v[j * 4 + j] + v[k * 4 + k]) + 1.0);
102 
103  T t;
104  if (s == 0.0) t = s;
105  else
106  t = static_cast<T>(0.5 / s);
107 
108  q[i] = s * 0.5f;
109  q[3] = (v[k * 4 + j] - v[j * 4 + k]) * t;
110  q[j] = (v[j * 4 + i] - v[i * 4 + j]) * t;
111  q[k] = (v[k * 4 + i] - v[i * 4 + k]) * t;
112  }
113 
114  scalar = q[3];
115  vec = {q[0], q[1], q[2]};
116  }

224  Mat as_matrix() const {

226  T y = vec[1];
227  T z = vec[2];

264  Vec as_euler() const {
265  Vec angles;
266 
267  // roll (x-axis rotation)
268  const T sinr_cosp = T(2) * (scalar * vec.x() + vec.y() * vec.z());
269  const T cosr_cosp = T(1) - T(2) * (vec.x() * vec.x() + vec.y() * vec.y());
270  angles[0] = Math::atan2(sinr_cosp, cosr_cosp);
271 
272  // yaw (y-axis rotation)
273  const T sinp = Math::sqrt(T(1) + T(2) * (scalar * vec.y() - vec.x() * vec.z()));
274  const T cosp = Math::sqrt(T(1) - T(2) * (scalar * vec.y() - vec.x() * vec.z()));
275  angles[1] = T(2) * Math::atan2(sinp, cosp) - T(Math::HALF_PI) / T(2);
276 
277  // pitch (z-axis rotation)
278  const T siny_cosp = T(2) * (scalar * vec.z() + vec.x() * vec.y());
279  const T cosy_cosp = T(1) - T(2) * (vec.y() * vec.y() + vec.z() * vec.z());
280  angles[2] = Math::atan2(siny_cosp, cosy_cosp);
281 
282  return angles;
283  }

334  static TQuat slerp(T t, const TQuat& a, const TQuat& b) {
335  T ang = angle(a, b);
336 
337  if (ang <= Math::THRESH_FLOAT32) {
338  return lerp(t, a, b);
339  }
340 
341  TQuat r;
342  T angleSin = Math::sin(ang);
343  T angle1 = Math::sin(ang * (1 - t)) / angleSin;
344  T angle2 = Math::sin(ang * t) / angleSin;
345 
346  r.scalar = a.scalar * angle1 + b.scalar * angle2;
347  r.vec = a.vec * angle1 + b.vec * angle2;
348 
349  return r;
350  }

382  static TQuat look_at(const TVecN<T, 3>& direction, const TVecN<T, 3>& up) {
383  auto Z = (-direction).normalized();
