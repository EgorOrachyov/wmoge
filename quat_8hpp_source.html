66  explicit TQuat(const Mat& mat) {
67  T q[4];// notation: x[0] y[1] z[2] w[3]
68  auto& v = mat.values;
69 
70  T trace = v[0] + v[5] + v[10];
71 
72  // Matrix 4x4 indexes
73  // 0 1 2 3
74  // 4 5 6 7
75  // 8 9 10 11
76  // 12 13 14 15
77 
78  // Check the diagonal
79  if (trace > 0.0) {
80  // positive diagonal
81 
82  T s = Math::sqrt(trace + 1.0f);
83  q[3] = static_cast<T>(s * 0.5);
84 
85  T t = 0.5f / s;
86  q[0] = (v[9] - v[6]) * t;
87  q[1] = (v[2] - v[8]) * t;
88  q[2] = (v[4] - v[1]) * t;
89  } else {
90  // negative diagonal
91 
92  int i = 0;
93  if (v[5] > v[0]) i = 1;
94  if (v[10] > v[4 * i + i]) i = 2;
95 
96  static const int NEXT[3] = {1, 2, 0};
97  int j = NEXT[i];
98  int k = NEXT[j];
99 
100  auto s = (T) Math::sqrt(v[i * 4 + i] - (v[j * 4 + j] + v[k * 4 + k]) + 1.0);
101 
102  T t;
103  if (s == 0.0) t = s;
104  else
105  t = static_cast<T>(0.5 / s);
106 
107  q[i] = s * 0.5f;
108  q[3] = (v[k * 4 + j] - v[j * 4 + k]) * t;
109  q[j] = (v[j * 4 + i] - v[i * 4 + j]) * t;
110  q[k] = (v[k * 4 + i] - v[i * 4 + k]) * t;
111  }
112 
113  scalar = q[3];
114  vec = {q[0], q[1], q[2]};
115  }

312  static TQuat slerp(T t, const TQuat& a, const TQuat& b) {
313  T ang = angle(a, b);
314 
315  if (ang <= Math::THRESH_FLOAT32) {
316  return lerp(t, a, b);
317  }
318 
319  TQuat r;
320  T angleSin = Math::sin(ang);
321  T angle1 = Math::sin(ang * (1 - t)) / angleSin;
322  T angle2 = Math::sin(ang * t) / angleSin;
