71  [[nodiscard]] std::array<TVecN<T, 3>, 8> vertices() const {
72  std::array<TVecN<T, 3>, 8> res;
73  res[0] = TVecN<T, 3>(pos.x() + size_half.x(), pos.y() + size_half.y(), pos.z() + size_half.z());
74  res[1] = TVecN<T, 3>(pos.x() + size_half.x(), pos.y() + size_half.y(), pos.z() - size_half.z());
75  res[2] = TVecN<T, 3>(pos.x() + size_half.x(), pos.y() - size_half.y(), pos.z() + size_half.z());
76  res[3] = TVecN<T, 3>(pos.x() + size_half.x(), pos.y() - size_half.y(), pos.z() - size_half.z());
77  res[4] = TVecN<T, 3>(pos.x() - size_half.x(), pos.y() + size_half.y(), pos.z() + size_half.z());
78  res[5] = TVecN<T, 3>(pos.x() - size_half.x(), pos.y() + size_half.y(), pos.z() - size_half.z());
79  res[6] = TVecN<T, 3>(pos.x() - size_half.x(), pos.y() - size_half.y(), pos.z() + size_half.z());
80  res[7] = TVecN<T, 3>(pos.x() - size_half.x(), pos.y() - size_half.y(), pos.z() - size_half.z());
81  return res;

84  [[nodiscard]] TAabb transform(const TMatMxN<T, 4, 4>& m) const {
85  std::array<TVecN<T, 3>, 8> v = vertices();
86 
87  for (int i = 0; i < 8; i++) {
88  v[i] = TVecN<T, 3>(m * TVecN<T, 4>(v[i].x(), v[i].y(), v[i].z(), T(1)));
89  }
90 
91  TAabb res;
92  res.pos = v[0];
93 
94  for (int i = 1; i < 8; i++) {
95  res = res.fit(v[i]);
96  }
97 
98  return res;
99  }
