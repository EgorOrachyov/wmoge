86  template<typename Component>
87  void EcsRegistry::register_component() {
88  const int component_idx = m_component_types++;
89  assert(component_idx < EcsLimits::MAX_COMPONENTS);
90 
91  EcsComponentInfo& component_info = m_components_info[component_idx];
92  assert(component_info.name == Strid());
93  assert(component_info.idx == -1);
94  assert(component_info.size == -1);
95 
96  Component::bind(component_idx, SID(Component::NAME_CSTR));
97 
98  component_info.name = Component::NAME;
99  component_info.idx = Component::IDX;
100  component_info.size = sizeof(Component);
101 
102  component_info.create = [](void* mem) -> void {
103  new (mem) Component();
104  };
105 
106  component_info.destroy = [](void* mem) -> void {
107  static_cast<Component*>(mem)->~Component();
108  };
109 
110  component_info.swap = [](void* mem1, void* mem2) -> void {
111  Component& cmp1 = *(static_cast<Component*>(mem1));
112  Component& cmp2 = *(static_cast<Component*>(mem2));
113  std::swap(cmp1, cmp2);
114  };
115 
116  m_components_name_to_idx[component_info.name] = component_info.idx;
117  m_components_pool[component_info.idx] = std::make_unique<MemPool>(component_info.size * m_chunk_size, m_expand_size);
118  }
