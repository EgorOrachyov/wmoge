109  Status set(GrcShaderParamId param_id, const T& v) {
110  if (param_id.is_invalid()) {
111  WG_LOG_ERROR("passed invalid param id");
112  return StatusCode::InvalidParameter;
113  }
114 
115  GrcShaderScript* script = provider.get_script();
116  std::optional<GrcShaderParamInfo*> p_info = script->get_param_info(param_id);
117 
118  if (!p_info) {
119  WG_LOG_ERROR("no such param id");
120  return StatusCode::InvalidParameter;
121  }
122 
123  GrcShaderParamInfo* param = p_info.value();
124  Ref<Data>* buffer = nullptr;
125 
126  if (param->buffer != -1) {
127  buffer = provider.get_buffer(param->space, param->buffer);
128  if (!buffer) {
129  WG_LOG_ERROR("no such buffer");
130  return StatusCode::InvalidState;
131  }
132 
133  const std::int16_t offset = param->offset;
134  const std::int16_t size = param->elem_count * param->type->byte_size;
135  std::uint8_t* ptr = (*buffer)->buffer();
136  T to_copy = v;
137 
138  if (size != sizeof(T)) {
139  WG_LOG_ERROR("mismatched param size to set");
140  return StatusCode::InvalidParameter;
141  }
142 
143  if constexpr (std::is_same_v<T, Mat2x2f>) {
144  to_copy = to_copy.transpose();
145  }
146  if constexpr (std::is_same_v<T, Mat3x3f>) {
147  to_copy = to_copy.transpose();
148  }
149  if constexpr (std::is_same_v<T, Mat4x4f>) {
150  to_copy = to_copy.transpose();
151  }
152 
153  std::memcpy(ptr + offset, (const std::uint16_t*) &to_copy, size);
154  return StatusCode::Ok;
155  }

158  if (!resources) {

160  return StatusCode::InvalidState;

164  GfxDescBindValue& bind_value = resource.second;
165

167  bind_value.resource = v.template as<GfxResource>();
168  return StatusCode::Ok;

172  return StatusCode::Ok;

179  bind_value.sampler = v;
