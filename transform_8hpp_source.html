130  void translate(const Vec3f& offset) { m_translation += offset; }

135  void scale(const Vec3f& scale) { m_scale *= scale; }

137  Transform3d inv() const {
138  Transform3d t;
139  t.set_translation(-m_translation);
140  t.set_rotation(m_rotation.conjugate());
141  t.set_scale(Vec3f(1.0f, 1.0f, 1.0f) / m_scale);
142  return t;
143  }
