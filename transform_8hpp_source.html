123  void rotate_x(float rad) { m_rotation = Quatf(Vec3f::axis_x(), rad) * m_rotation; }
124  void rotate_y(float rad) { m_rotation = Quatf(Vec3f::axis_y(), rad) * m_rotation; }

126  void scale(const Vec3f& scale) { m_scale *= scale; }

128  Vec3f transform(const Vec3f& v) const {
129  return m_translation + m_rotation.rotate(m_scale * v);
130  }
131 
132  Transform3d operator*(const Transform3d& other) const {
133  // v' = t1 + r1(s1 * v)
134  // v'' = t2 + r2(s2 * v')
135  // = t2 + r2(s2 * t1 + s2 * r1(s1 * v))
136  // = (t2 + r2(s2 * t1)) + r2(s2 * r1(s1 * v))
137  // = (t2 + r2(s2 * t1)) + r2(r1(r1^-1(s2) * s1 * v)))
138  //
139  // t' = t2 + r2(s2 * t1)
140  // r' = r2 x r1
141  // s' = r1^-1(s2) * s1
142 
143  Transform3d result;
144  result.set_translation(m_translation + transform(other.m_translation));
145  result.set_rotation(m_rotation * other.m_rotation);
146  result.set_scale(other.m_rotation.conjugate().rotate(m_scale) * other.m_scale);
147  return result;
148  }
