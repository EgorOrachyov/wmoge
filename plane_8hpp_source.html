50  TPlane(const TVecN<T, 3>& dir, const TVecN<T, 3>& pos) {
51  // dot(x - pos, norm) == 0
52  // dot(x,norm) - dot(pos,norm) == 0
53  // dot(x,norm) - d == 0
54  norm = dir.normalized();
55  d = TVecN<T, 3>::dot(pos, norm);
56  }
57 
58  TPlane(const TVecN<T, 3>& x, const TVecN<T, 3>& y, const TVecN<T, 3>& z) {
59  // |
60  // x|_____z
61  // /
62  // y/ plane, facing top
63  TVecN<T, 3> pos = x;
64  TVecN<T, 3> dir = TVecN<T, 3>::cross(y - x, z - x);
65  norm = dir.normalized();
66  d = TVecN<T, 3>::dot(pos, norm);
67  }
68 
69  T dot(const TVecN<T, 3>& x) const {
70  return TVecN<T, 3>::dot(x, norm) - d;
71  }
72 
73  bool check(const TVecN<T, 3>& x) {
74  return TVecN<T, 3>::dot(x, norm) - d >= static_cast<T>(0);
75  }
76  bool check_strict(const TVecN<T, 3>& x) {
77  return TVecN<T, 3>::dot(x, norm) - d > static_cast<T>(0);
78  }
