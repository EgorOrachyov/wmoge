71  Status tree_read(IoContext& context, IoTree& tree, flat_set<K>& set) {
72  assert(set.empty());
73  set.reserve(tree.node_num_children());
74  tree.node_find_first_child();
75  for (; tree.node_is_valid(); tree.node_next_sibling()) {
