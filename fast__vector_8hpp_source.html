81  Status yaml_read(const YamlConstNodeRef& node, fast_vector<T, MinCapacity>& vector) {
82  assert(vector.empty());
83  vector.resize(node.num_children());
84  std::size_t element_id = 0;
85  for (auto child = node.first_child(); child.valid(); child = child.next_sibling()) {
86  WG_YAML_READ(child, vector[element_id]);
87  element_id += 1;
88  }
89  return StatusCode::Ok;
